    main.cpp

    audioplayer.cpp
    recording.cpp

    audioprober.cpp
    utils.cpp
    settingsmodel.cpp
)

add_executable(krecorder ${krecorder_SRCS} ${RESOURCES})
target_link_libraries(krecorder 
    Qt5::Core 
    Qt5::Gui 
    Qt5::Qml 
    Qt5::Quick 
    Qt5::Widgets 
    Qt5::QuickControls2 
    Qt5::Multimedia
    Qt5::Svg
    KF5::I18n
    KF5::ConfigCore
    KF5::ConfigGui
    KF5::CoreAddons
)
